  <script>
    const btnOui = document.getElementById('oui');
    const btnNon = document.getElementById('non');
    const message = document.getElementById('message');
    const body = document.body;
    const musique = document.getElementById('musique');

    btnNon.style.left = "120px";

    // Clique sur "Oui"
    btnOui.addEventListener('click', () => {
      message.textContent = "LESGOO cette fois on fait ça correctement ! 💞";
      message.classList.add('show');
      createHearts();

      musique.volume = 0.5;
      musique.play();
    });

    // Le bouton "Non" fuit
    btnNon.addEventListener('mouseover', () => {
      const maxX = window.innerWidth - btnNon.offsetWidth;
      const maxY = window.innerHeight - btnNon.offsetHeight;
      const newX = Math.random() * maxX - maxX / 2;
      const newY = Math.random() * maxY - maxY / 2;
      btnNon.style.transition = "transform 0.05s ease";
      btnNon.style.transform = `translate(${newX}px, ${newY}px)`;
    });

    // Si quelqu’un clique sur "Non"
    btnNon.addEventListener('click', () => {
      message.textContent = "T’as une meilleure aim que sur Valo 😅";
      message.classList.add('show');
    });

    // Cœurs 💕
    function createHearts() {
      for (let i = 0; i < 25; i++) {
        const heart = document.createElement('div');
        heart.classList.add('heart');
        heart.textContent = '💖';
        heart.style.left = Math.random() * window.innerWidth + 'px';
        heart.style.bottom = '-20px';
        heart.style.animationDuration = 2.5 + Math.random() * 2 + 's';
        heart.style.fontSize = 1 + Math.random() * 1.5 + 'em';
        body.appendChild(heart);
        setTimeout(() => heart.remove(), 5000);
      }
    }
  </script>
